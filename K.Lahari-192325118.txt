#Selection sort:
#include<stdio.h>
void selection_sort(int a[],int n)
{
	int i,j,temp;
	for(i=0;i<n-1;i++)
	{
		int min=i;
		for(j=i+1;j<n;j++)
		{
			if (a[j]<a[min])
			{
				min=j;
			}
		}
		if(min!=i)
		{
			temp=a[i];
			a[i]=a[min];
			a[min]=temp;
		}
	}
}
void print_array(int a[],int n)
{
	for (int i=0;i<n;i++){
		printf(" %d ",a[i]);
	}
	printf("\n");
}
int main()
{
	int a[]={23,45,2,156,68,899,76};
	int n=sizeof(a)/sizeof(a[0]);
	printf("Sorted Array:");
	print_array(a,n);
	selection_sort(a,n);
	printf("Unsorted Array:");
	print_array(a,n);
	return 0;
}
Output:
Sorted Array: 23  45  2  156  68  899  76
Unsorted Array: 2  23  45  68  76  156  899
